323    def get_stat(self, start_date: str, end_date: str) -> pd.DataFrame:
324        """METHOD COMPUTING ANNUAL RETURNS, ANNUAL STD. DEV. & SHARPE RATIO OF ASSETS"""
325
326        # ANALYZE THE DATA for a given time period
327        weekly_data = self.weeklyReturns[
328            (self.weeklyReturns.index >= start_date) & (self.weeklyReturns.index <= end_date)
329        ].copy()
330
331        # Create table with summary statistics
332        mu_ga = mean_an_returns(weekly_data)  # Annualised geometric mean of returns
333        std_dev_a = weekly_data.std(axis=0) * np.sqrt(52)  # Annualised standard deviation of returns
334        sharpe = round(mu_ga / std_dev_a, 2)  # Sharpe ratio of each financial product
335
336        # Write all results into a data frame
337        stat_df = pd.concat([mu_ga, std_dev_a, sharpe], axis=1)
338        stat_df.columns = [
339            "Average Annual Returns",
340            "Standard Deviation of Returns",
341            "Sharpe Ratio",
342        ]
343        stat_df["ISIN"] = stat_df.index  # Add names into the table
344        stat_df["Name"] = self.names
345        stat_df["Size"] = 1
346        stat_df["Type"] = "ETF"
347
348        return stat_df

350    def get_top_performing_assets(self, time_periods: List[Tuple[str, str]], top_percent: float = 0.2) -> List[str]:
351        stats_for_periods = {f"period_{i}": self.get_stat(*period) for i, period in enumerate(time_periods, 1)}
352
353        # Create 'Risk class' column where the value is
354        # 'Risk Class 1' if Standard Deviation of Returns <= 0.005
355        # 'Risk Class 2' if > 0.005 and < 0.02
356        # 'Risk Class 3' if > 0.02 and < 0.05
357        # 'Risk Class 4' if > 0.05 and < 0.1
358        # 'Risk Class 5' if > 0.1 and < 0.15
359        # 'Risk Class 6' if > 0.15 and < 0.25 then
360        # 'Risk Class 7' if > 0.25
361        risk_level = {
362            "Risk Class 1": 0.005,
363            "Risk Class 2": 0.02,
364            "Risk Class 3": 0.05,
365            "Risk Class 4": 0.10,
366            "Risk Class 5": 0.15,
367            "Risk Class 6": 0.25,
368            "Risk Class 7": 1,
369        }
370        for data in stats_for_periods.values():
371            data["Risk Class"] = pd.cut(
372                data["Standard Deviation of Returns"],
373                bins=[-1] + list(risk_level.values()),
374                labels=list(risk_level.keys()),
375                right=True,
376            )
377        # For each data_period and each risk class, find the top 20% best performing assets
378        # mark them as True in column 'Top Performer'
379        for data in stats_for_periods.values():
380            for risk_class in risk_level.keys():
381                data.loc[
382                    data["Risk Class"] == risk_class,
383                    "Top Performer",
384                ] = data.loc[data["Risk Class"] == risk_class, "Sharpe Ratio"].rank(pct=True) > (1 - top_percent)
385        # for each period, save the pandas dataframe into excel files
386        # for index, data in enumerate(stats_for_periods.values()):
387        #     data.to_excel(f"top_performers_{time_periods[index]}.xlsx")
388
389        # ISIN codes for assets which were top performers in all n periods
390        top_isins = stats_for_periods["period_1"].loc[stats_for_periods["period_1"]["Top Performer"], "ISIN"].values
391        for data in stats_for_periods.values():
392            top_isins = np.intersect1d(top_isins, data.loc[data["Top Performer"], "ISIN"].values)
393
394        top_names = [self.names[self.tickers.index(isin)] for isin in top_isins]
395
396        return top_names

324    def get_stat(self, start_date: str, end_date: str) -> pd.DataFrame:
325        """METHOD COMPUTING ANNUAL RETURNS, ANNUAL STD. DEV. & SHARPE RATIO OF ASSETS"""
326

351    def get_top_performing_assets(self, time_periods: List[Tuple[str, str]], top_percent: float = 0.2) -> List[str]:
352        stats_for_periods = {f"period_{i}": self.get_stat(*period) for i, period in enumerate(time_periods, 1)}
353
354        # Create 'Risk class' column where the value is
355        # 'Risk Class 1' if Standard Deviation of Returns <= 0.005
356        # 'Risk Class 2' if > 0.005 and < 0.02
357        # 'Risk Class 3' if > 0.02 and < 0.05
358        # 'Risk Class 4' if > 0.05 and < 0.1
359        # 'Risk Class 5' if > 0.1 and < 0.15
360        # 'Risk Class 6' if > 0.15 and < 0.25 then
361        # 'Risk Class 7' if > 0.25
362        risk_level = {
363            "Risk Class 1": 0.005,
364            "Risk Class 2": 0.02,
365            "Risk Class 3": 0.05,
366            "Risk Class 4": 0.10,
367            "Risk Class 5": 0.15,
368            "Risk Class 6": 0.25,
369            "Risk Class 7": 1,
370        }
371        for data in stats_for_periods.values():
372            data["Risk Class"] = pd.cut(
373                data["Standard Deviation of Returns"],
374                bins=[-1] + list(risk_level.values()),
375                labels=list(risk_level.keys()),
376                right=True,
377            )
378        # For each data_period and each risk class, find the top 20% best performing assets
379        # mark them as True in column 'Top Performer'
380        for data in stats_for_periods.values():
381            for risk_class in risk_level.keys():
382                data.loc[
383                    data["Risk Class"] == risk_class,
384                    "Top Performer",
385                ] = data.loc[data["Risk Class"] == risk_class, "Sharpe Ratio"].rank(pct=True) > (1 - top_percent)
386        # for each period, save the pandas dataframe into excel files

392        for data in stats_for_periods.values():
